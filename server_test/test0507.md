```bash

  -H "Content-Type: application/json" \
  -d '{
    "invoice_number": "test-aero-001",
    "invoice_date": "2026-04-28",
    "currency": "USD",
    "total_amount": 1.0,
    "parties": [
      {
        "role": "BILL_TO",
        "name": "AEROCARIBBEAN AIRLINES",
        "address": "",
        "country": "",
        "registration_number": "",
        "tax_id": "",
        "account_number": "",
        "swift": "",
        "iban": "",
        "account_holder": "",
        "phone": "",
        "email": ""

    ]
  }' ; echo
```
